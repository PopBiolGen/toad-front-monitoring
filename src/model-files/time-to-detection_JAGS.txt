# time to detection model in JAGS
model{
  # priors
  p.occ ~ dunif(0, 1)
  lambda ~ dunif(0, 1)
  
  for (ii in 1:n.obs){
    ttd[ii] ~ dexp(lambda) # time to detection
    occ[ii] ~ dbern(p.occ) # latent variable. present at site?
    censored[ii] ~ dbern(occ[ii]*step(ttd[ii] - tmax.i[ii]) + (1-occ[ii])) # model for non-detections
  }
}
