# time to detection model in JAGS
# infers current location of the invasion front
model{
  # priors
  lambda ~ dunif(0, 1)
  beta ~ dunif(0,4) # sensible prior given logit space
  a ~ dnorm(0, 1e-8) # very flat given scale is in metres
  b ~ dnorm(0, 1e-6) # could do better on this given history of invasion, but uninformative for now
  
  for (ii in 1:n.obs){
    # time to detection
    ttd[ii] ~ dexp(lambda) # time to detection, for sites with a detection
    
    # occupancy
    occ[ii] ~ dbern(p.occ[ii]) # latent variable. Present at site?
    logit(p.occ[ii]) <- beta * d[ii] #occupancy as a function of distance from invasion front
    d[ii] <- (y[ii]-a*x[ii]-b)/sqrt(1+a^2) # perpendicular distance to line at t[ii]
    
    # detected, or not, given occupancy and inferred distribution of time to detection
    is.detected[ii] ~ dbern(p.detect[ii]) # detected, or not
    p.detect[ii] <- occ[ii]*(1-exp(-lambda*tmax.i[ii])) # probability of detection
  }
}
